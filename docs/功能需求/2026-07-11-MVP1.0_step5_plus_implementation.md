```python
"""BIA / PPG → 标准指标字典 提取器。

设计依据: docs/planning/MVP1.0_技术方案_v1.1.md §6.2 / §18 Step 2

将硬件层 dataclass / dict 映射到 ``medical_knowledge.json`` 指标 id。
下游 :class:`MedicalKnowledgeBase.judge()` / :class:`HealthAssessmentPipeline`
统一消费 ``indicators: dict[str, float | None]``。
"""
from __future__ import annotations

from typing import Any, Optional


# BIA 字段映射表（硬件 dict key → 医学知识库指标 id）
_BIA_FIELD_MAP = {
    "体重_kg": "weight_kg",
    "体脂率_%": "body_fat_pct",
    "水分率_%": "water_pct",
    "肌肉量_kg": "muscle_kg",
    "骨骼肌_kg": "skeletal_muscle_kg",
    "基础代谢_kcal": "bmr_kcal",
    "骨量_kg": "bone_kg",
    "内脏脂肪等级": "visceral_fat_level",
    "皮下脂肪_%": "subcutaneous_fat_pct",
    "身体年龄_岁": "body_age",
    "BMI": "bmi",
}

# PPG 字段映射（dataclass attr → 指标 id）
_PPG_FIELD_MAP = {
    "heart_rate": "heart_rate",
    "spo2": "spo2",
    "pi_percent": "perfusion_index",
    "hrv_ms": "hrv_rmssd",
}


def extract_from_bia(body_composition: Optional[dict[str, Any]]) -> dict[str, Any]:
    """从 BIA 体成分 dict 中提取标准指标。

    缺失字段返回 ``None``（不要默认填 0，避免误导 knowledge base 判定）。
    """
    out: dict[str, Any] = {}
    if not body_composition:
        return out

    for hw_key, ind_id in _BIA_FIELD_MAP.items():
        if hw_key in body_composition:
            out[ind_id] = body_composition[hw_key]

    # 蛋白质率需要从蛋白质_kg / 体重_kg 推导
    if "蛋白质_kg" in body_composition and "体重_kg" in body_composition:
        weight = body_composition["体重_kg"]
        if weight and weight > 0:
            out["protein_pct"] = round(body_composition["蛋白质_kg"] / weight * 100, 2)
        else:
            out["protein_pct"] = None
    return out


def extract_from_ppg(reading: Optional["PPGReading"]) -> dict[str, Any]:  # noqa: F821
    """从 PPGReading dataclass 提取指标。

    None 输入返回空 dict（PPG 不可用场景）。
    """
    out: dict[str, Any] = {}
    if reading is None:
        return out
    for attr, ind_id in _PPG_FIELD_MAP.items():
        if hasattr(reading, attr):
            out[ind_id] = getattr(reading, attr)
    return out


def merge_indicators(
    bia: dict[str, Any],
    ppg: dict[str, Any],
    user: dict[str, Any],
) -> dict[str, Any]:
    """合并 BIA + PPG + 派生指标。

    派生:
      * ``bmi`` = weight / height²
      * ``smmi`` = skeletal_muscle / height²
      * ``metabolic_age_delta`` = body_age - user_age
    """
    out = {**bia, **ppg}

    height_m = (user.get("height_cm") or 0) / 100.0
    weight = bia.get("weight_kg")
    if weight and height_m > 0 and "bmi" not in out:
        out["bmi"] = round(weight / (height_m ** 2), 2)

    skm = bia.get("skeletal_muscle_kg")
    if skm and height_m > 0:
        out["smmi"] = round(skm / (height_m ** 2), 2)

    body_age = bia.get("body_age")
    user_age = user.get("age")
    if body_age is not None and user_age is not None:
        out["metabolic_age_delta"] = int(body_age) - int(user_age)

    return out


__all__ = ["extract_from_bia", "extract_from_ppg", "merge_indicators"]
```

- [ ] **Step 5: 实现 `src/pipeline/health_assessment_pipeline.py`**

```python
"""端到端健康评估管线 (v2 — UI-ready)。

按 §18 6-step 模型:

  Step 1: 硬件采集 (BIA + PPG)         via BIAModule / PPGModule
  Step 2: 字段提取                       via biometric_extractors
  Step 3: 知识库判定                     via MedicalKnowledgeBase
  Step 4: 引擎触发 (advice/emotion/alert/composite)  via 4 engines
  Step 5: 渲染/模板                      via ReportEngine
  Step 6: 持久化 + FinalReport 组装      via DBManager

设计要点:
    * 所有引擎通过依赖注入, 测试可替换 mock
    * 默认 from_default() 使用真实串口配置 (生产) 或 MockTransport (测试)
    * 异常隔离: 单个引擎失败不阻断管线, 错误累积到 result.errors
    * persist_to_db=False 时跳过入库 (用于无 DB 场景)
"""
from __future__ import annotations

import logging
from dataclasses import asdict
from datetime import datetime
from pathlib import Path
from typing import Any, Optional

from src.config import AppConfig, load_config
from src.indicators.biometric_extractors import (
    extract_from_bia,
    extract_from_ppg,
    merge_indicators,
)
from src.knowledge.advice_engine import AdviceEngine, AdviceItem, MANDATORY_DISCLAIMER
from src.knowledge.composite_score_engine import CompositeScoreEngine
from src.knowledge.emotion_analyzer import EmotionAnalyzer
from src.knowledge.medical_db import MedicalKnowledgeBase
from src.knowledge.medical_alert_engine import MedicalAlertEngine
from src.knowledge.report_engine import ReportEngine
from src.knowledge.transformation_pipeline import TransformationPipeline
from src.pipeline.report_renderer import (
    AlertEntry,
    CompositeScoreEntry,
    EmotionalStateLabel,
    FinalReport,
    HealthIndicators,
    PipelineResult,
    ReportTemplateEntry,
    RiskLevel,
)

logger = logging.getLogger(__name__)


class HealthAssessmentPipeline:
    """MVP1.0 端到端编排器。

    使用:
        pipeline = HealthAssessmentPipeline.from_default()
        report = pipeline.run(user_dict, pss4_score=5)
        print(report.disclaimer)
        for a in report.advices:
            print(a.content)
    """

    def __init__(
        self,
        config: AppConfig,
        bia_module=None,
        ppg_module=None,
        kb: Optional[MedicalKnowledgeBase] = None,
        advice_engine: Optional[AdviceEngine] = None,
        emotion_analyzer: Optional[EmotionAnalyzer] = None,
        alert_engine: Optional[MedicalAlertEngine] = None,
        composite_engine: Optional[CompositeScoreEngine] = None,
        report_engine: Optional[ReportEngine] = None,
        transformation: Optional[TransformationPipeline] = None,
        db_manager=None,
    ) -> None:
        self.config = config
        self.bia_module = bia_module
        self.ppg_module = ppg_module
        self.kb = kb or MedicalKnowledgeBase()
        self.advice_engine = advice_engine or AdviceEngine(knowledge_base=self.kb)
        self.emotion_analyzer = emotion_analyzer or EmotionAnalyzer()
        self.alert_engine = alert_engine or MedicalAlertEngine()
        self.composite_engine = composite_engine or CompositeScoreEngine()
        self.report_engine = report_engine or ReportEngine()
        self.transformation = transformation or TransformationPipeline()
        self.db_manager = db_manager  # 延迟注入

    # ------------------------------------------------------------------ #
    # 工厂
    # ------------------------------------------------------------------ #

    @classmethod
    def from_default(cls) -> "HealthAssessmentPipeline":
        """基于当前配置构造（使用真实 BIA/PPG 串口）。"""
        return cls(config=load_config())

    # ------------------------------------------------------------------ #
    # 主流程
    # ------------------------------------------------------------------ #

    def run(
        self,
        user: dict[str, Any],
        pss4_score: Optional[int] = None,
        history: Optional[list[dict]] = None,
        persist_to_db: bool = False,
        user_id: Optional[int] = None,
    ) -> FinalReport:
        """执行端到端流程。"""
        result = PipelineResult()

        try:
            self._step1_acquire(user, result)
        except Exception as e:  # noqa: BLE001
            logger.warning("Step 1 测量失败, 继续使用空数据: %s", e)
            result.errors.append(f"step1: {e}")

        try:
            self._step2_extract(user, result)
        except Exception as e:  # noqa: BLE001
            logger.exception("Step 2 提取失败")
            result.errors.append(f"step2: {e}")

        try:
            self._step3_judge(user, result)
        except Exception as e:  # noqa: BLE001
            logger.exception("Step 3 判定失败")
            result.errors.append(f"step3: {e}")

        try:
            self._step4_engines(user, pss4_score, history, result)
        except Exception as e:  # noqa: BLE001
            logger.exception("Step 4 引擎失败")
            result.errors.append(f"step4: {e}")

        try:
            self._step5_render(user, result)
        except Exception as e:  # noqa: BLE001
            logger.exception("Step 5 渲染失败")
            result.errors.append(f"step5: {e}")

        measurement_id = None
        if persist_to_db and self.db_manager is not None and user_id is not None:
            try:
                measurement_id = self._step6_persist(
                    user, user_id, result
                )
            except Exception as e:  # noqa: BLE001
                logger.exception("Step 6 持久化失败")
                result.errors.append(f"step6: {e}")

        return self._assemble_final_report(user, result, measurement_id, user_id)

    # ------------------------------------------------------------------ #
    # Step 1: 测量
    # ------------------------------------------------------------------ #

    def _step1_acquire(self, user: dict, result: PipelineResult) -> None:
        """BIA + PPG 数据采集（若注入 mock 则用 mock）。"""
        if self.bia_module is not None:
            try:
                bia_reading = self.bia_module.measure(user)
                result.raw_bia = bia_reading.body_composition
            except Exception as e:  # noqa: BLE001
                logger.warning("BIA 测量失败: %s", e)
                result.raw_bia = None
        else:
            result.raw_bia = None

        if self.ppg_module is not None:
            try:
                readings = self.ppg_module.collect(duration_sec=15)
                result.raw_ppg = [
                    {
                        "spo2": r.spo2, "heart_rate": r.heart_rate,
                        "perfusion_index": r.pi_percent, "hrv_rmssd": r.hrv_ms,
                        "valid": r.valid,
                    }
                    for r in readings if r.valid
                ]
            except Exception as e:  # noqa: BLE001
                logger.warning("PPG 采集失败: %s", e)
                result.raw_ppg = None
        else:
            result.raw_ppg = None

    # ------------------------------------------------------------------ #
    # Step 2: 提取
    # ------------------------------------------------------------------ #

    def _step2_extract(self, user: dict, result: PipelineResult) -> None:
        """硬件原始 → 标准指标 dict。"""
        bia_dict = extract_from_bia(result.raw_bia)
        ppg_dict = {}
        if result.raw_ppg:
            # 取均值（或首条）作为 PPG 标准指标
            from src.hardware.ppg_module import PPGReading
            r = PPGReading(
                status=2,
                spo2=int(sum(p["spo2"] for p in result.raw_ppg) / len(result.raw_ppg)),
                heart_rate=int(sum(p["heart_rate"] for p in result.raw_ppg) / len(result.raw_ppg)),
                pi_percent=sum(p["perfusion_index"] for p in result.raw_ppg) / len(result.raw_ppg),
                hrv_ms=int(sum(p["hrv_rmssd"] for p in result.raw_ppg) / len(result.raw_ppg)),
                pulse_height=120, systole_time_ms=380,
                adc_ir=1500, adc_red=1300, valid=True,
            )
            ppg_dict = extract_from_ppg(r)

        result.indicators = merge_indicators(bia_dict, ppg_dict, user)

    # ------------------------------------------------------------------ #
    # Step 3: 知识库判定
    # ------------------------------------------------------------------ #

    def _step3_judge(self, user: dict, result: PipelineResult) -> None:
        """每项指标调用 KB.judge, 得到 status dict。"""
        age = user.get("age", 30)
        gender = user.get("gender", 1)
        for ind_id, value in result.indicators.items():
            if value is None:
                continue
            try:
                status = self.kb.judge(ind_id, value, age, gender)
                result.statuses[ind_id] = status.value
            except Exception:  # noqa: BLE001
                logger.debug("KB judge 跳过 %s", ind_id)

    # ------------------------------------------------------------------ #
    # Step 4: 引擎 (advice / emotion / alert / composite)
    # ------------------------------------------------------------------ #

    def _step4_engines(self, user, pss4_score, history, result: PipelineResult) -> None:
        # 4a. Advice (三层 + 强免责)
        advices = self.advice_engine.generate(
            result.indicators, user, pss4_score=pss4_score, history=history
        )
        result.advices = advices

        # 4b. Emotion (VAD + 标签 + PSS-4 融合)
        emo = self.emotion_analyzer.analyze(
            rmssd=result.indicators.get("hrv_rmssd"),
            resting_hr=result.indicators.get("heart_rate"),
            stress_idx=result.indicators.get("stress_index"),
            pss4_score=pss4_score,
        )
        result.emotional_state_label = EmotionalStateLabel(emo.label)
        result.emotional_confidence = emo.confidence

        # 4c. Medical Alert (按 medical_alert.json 规则)
        alerts = self.alert_engine.evaluate(
            indicators=result.indicators, user=user,
            confidence=emo.confidence,
        )
        result.alerts = [
            AlertEntry(rule_id=a.id, level=a.level, message=a.message, medical_required=a.medical_required)
            for a in alerts
        ]

        # 4d. Composite Scores (代谢/免疫/心肺/压力/骨骼)
        scores = self.composite_engine.calculate_all(
            indicators=result.indicators, user=user
        )
        result.composite_scores = [
            CompositeScoreEntry(
                score_id=s.id, name=s.name, value=s.value,
                interpretation=s.interpretation,
                components=s.components,
            )
            for s in scores
        ]

    # ------------------------------------------------------------------ #
    # Step 5: 渲染 (12 条模板 → ReportTemplateEntry)
    # ------------------------------------------------------------------ #

    def _step5_render(self, user: dict, result: PipelineResult) -> None:
        rendered = self.report_engine.render_all(
            indicators=result.indicators, user=user,
            advices=result.advices, alerts=result.alerts,
            composite_scores=result.composite_scores,
        )
        result.templates = [
            ReportTemplateEntry(template_id=t.id, title=t.title, sections=t.sections)
            for t in rendered
        ]

    # ------------------------------------------------------------------ #
    # Step 6: 持久化
    # ------------------------------------------------------------------ #

    def _step6_persist(self, user: dict, user_id: int, result: PipelineResult) -> Optional[int]:
        """写入 measurement + advice 表。"""
        from src.database.models import Advice, Measurement, dumps_raw

        # 转 numpy/int/float 为基础类型
        m_dict = {k: (float(v) if hasattr(v, "item") else v) for k, v in result.indicators.items()}
        m = Measurement(
            user_id=user_id,
            weight_kg=m_dict.get("weight_kg"),
            body_fat_pct=m_dict.get("body_fat_pct"),
            water_pct=m_dict.get("water_pct"),
            protein_pct=m_dict.get("protein_pct"),
            muscle_kg=m_dict.get("muscle_kg"),
            skeletal_muscle_kg=m_dict.get("skeletal_muscle_kg"),
            bmr_kcal=int(m_dict.get("bmr_kcal") or 0) or None,
            bone_kg=m_dict.get("bone_kg"),
            visceral_fat_level=int(m_dict.get("visceral_fat_level") or 0) or None,
            body_age=int(m_dict.get("body_age") or 0) or None,
            heart_rate=int(m_dict.get("heart_rate") or 0) or None,
            spo2=int(m_dict.get("spo2") or 0) or None,
            perfusion_index=m_dict.get("perfusion_index"),
            hrv_rmssd=m_dict.get("hrv_rmssd"),
        )
        mid = self.db_manager.create_measurement(m)

        advice_rows = [
            Advice(
                measurement_id=mid,
                category=a.category,
                priority=a.priority,
                content=a.content + "\n\n" + MANDATORY_DISCLAIMER,
            )
            for a in result.advices
        ]
        self.db_manager.bulk_create_advice(advice_rows)
        return mid

    # ------------------------------------------------------------------ #
    # Final 报告组装
    # ------------------------------------------------------------------ #

    def _assemble_final_report(
        self, user: dict, result: PipelineResult,
        measurement_id: Optional[int], user_id: Optional[int],
    ) -> FinalReport:
        hi = HealthIndicators(**{k: v for k, v in result.indicators.items() if k in HealthIndicators.__dataclass_fields__})

        # 综合健康分 = 综合分均值 (用于风险等级)
        avg_score = (
            sum(s.value for s in result.composite_scores) / len(result.composite_scores)
            if result.composite_scores else 70.0
        )
        hi.composite_health_score = int(round(avg_score))

        risk_level = self._derive_risk_level(result, avg_score)

        return FinalReport(
            health_indicators=hi,
            emotional_state=result.emotional_state_label,

            advices=result.advices,
            risk_level=risk_level,
            composite_scores=result.composite_scores,
            alerts=result.alerts,
            templates=result.templates,
            disclaimer=MANDATORY_DISCLAIMER,
            measurement_id=measurement_id,
            user_id=user_id,
            report_at=datetime.now().isoformat(timespec="seconds"),
        )

    def _derive_risk_level(self, result: PipelineResult, avg_score: float) -> RiskLevel:
        if any(a.priority == "urgent" for a in result.advices):
            return RiskLevel.URGENT
        if any(a.level == "red" for a in result.alerts):
            return RiskLevel.HIGH
        if avg_score < 50:
            return RiskLevel.MEDIUM
        if avg_score < 70:
            return RiskLevel.MEDIUM
        return RiskLevel.LOW
```

```python
def _stage3_assess(self, config: dict, data: dict, user: dict) -> dict:
    """Stage 3: 健康评估 — 调用 MedicalKnowledgeBase 批量判定。"""
    from src.knowledge.medical_db import MedicalKnowledgeBase

    kb = MedicalKnowledgeBase()
    statuses = {}
    assessment_errors = []
    age = user.get("age", 30)
    gender = user.get("gender", 1)
    for ind_id, value in data.items():
        if value is None:
            continue
        try:
            status = kb.judge(ind_id, float(value), int(age), int(gender))
            statuses[ind_id] = status.value
        except Exception as e:  # noqa: BLE001
            assessment_errors.append(f"{ind_id}: {e}")

    return {
        "statuses": statuses,
        "assessment_errors": assessment_errors,
    }


def _stage4_report(self, config: dict, data: dict, user: dict) -> dict:
    """Stage 4: 报告生成 — 委托 ReportEngine + AdviceEngine + AlertEngine + CompositeScoreEngine。"""
    from src.knowledge.advice_engine import AdviceEngine
    from src.knowledge.composite_score_engine import CompositeScoreEngine
    from src.knowledge.emotion_analyzer import EmotionAnalyzer
    from src.knowledge.medical_alert_engine import MedicalAlertEngine
    from src.knowledge.report_engine import ReportEngine

    # 1. 引擎触发
    emotion = EmotionAnalyzer().analyze(
        rmssd=data.get("hrv_rmssd"),
        resting_hr=data.get("heart_rate"),
        stress_idx=data.get("stress_index"),
        pss4_score=data.get("pss4_score"),
    )
    advice_items = AdviceEngine().generate(data, user, pss4_score=data.get("pss4_score"))
    alerts = MedicalAlertEngine().evaluate(indicators=data, user=user, confidence=emotion.confidence)
    scores = CompositeScoreEngine().calculate_all(indicators=data, user=user)

    # 2. 模板渲染
    rendered = ReportEngine().render_all(
        indicators=data, user=user, advices=advice_items,
        alerts=alerts, composite_scores=scores,
    )

    return {
        "emotion": {"label": emotion.label, "confidence": emotion.confidence},
        "advices": [a.rule_id for a in advice_items],
        "alerts": [{"id": a.id, "level": a.level, "message": a.message} for a in alerts],
        "scores": [{"id": s.id, "value": s.value, "interpretation": s.interpretation} for s in scores],
        "templates": [{"id": t.id, "title": t.title, "sections": t.sections} for t in rendered],
    }
```

```python
PSS-4 等级映射: 0-4低 / 5-8中 / 9-12较高 / 13-16高
"""
from __future__ import annotations

import json

import re
from dataclasses import dataclass, field

from src.config import KNOWLEDGE_DIR

logger = logging.getLogger(__name__)


@dataclass
class EmotionResult:
    """情绪分析结果。"""
    label: str
    vad: dict[str, int] = field(default_factory=dict)
    templates: list[str] = field(default_factory=list)
    confidence: float = 0.0
    medical_required: bool = False
    disclaimer: str = "此分类为产品自定义规则，非临床心理学评估工具，仅供参考。"


def _eval_condition(cond: str, ctx: dict[str, Any]) -> bool:
    """安全地评估 emotion_model.json 中的 condition 表达式。

    支持: 比较、and/or、恒等比较、simple numeric range.
    使用 Python AST 白名单求值器 (复用 src.utils.safe_condition_evaluator).
    """
    from src.utils.safe_condition_evaluator import SafeConditionEvaluator
    try:
        return SafeConditionEvaluator().evaluate(cond, ctx)
    except Exception:  # noqa: BLE001
        logger.debug("emotion condition 评估失败: %s", cond)
        return False


class EmotionAnalyzer:
    """情绪分析器 — VAD 维度 + 模板匹配。"""

    def __init__(self, model_path: Optional[Path] = None) -> None:
        self.model_path = model_path or (KNOWLEDGE_DIR / "emotion_model.json")
        if not self.model_path.exists():
            raise FileNotFoundError(f"emotion_model.json 不存在: {self.model_path}")
        with self.model_path.open(encoding="utf-8") as f:
            self.model = json.load(f)
        self.pss4_mapping = self.model["pss4_mapping"]
        self.templates = self.model["emotion_templates"]

    def analyze(
        self,
        rmssd: Optional[float] = None,
        resting_hr: Optional[int] = None,
        stress_idx: Optional[float] = None,
        pss4_score: Optional[int] = None,
    ) -> EmotionResult:
        """分析情绪状态, 返回 VAD + 模板列表。

        Args:
            rmssd: HRV RMSSD (ms), None 视作充足休息状态
            resting_hr: 静息心率 (bpm)
            stress_idx: 压力指数 (0-100)
            pss4_score: PSS-4 问卷总分 (0-16), None 视为 0
        """
        pss4 = pss4_score if pss4_score is not None else 0

        # 1) 决定状态标签 + VAD
        if pss4 <= 4:
            bucket = "0-4"
        elif pss4 <= 8:
            bucket = "5-8"
        elif pss4 <= 12:
            bucket = "9-12"
        else:
            bucket = "13-16"

        mapping = self.pss4_mapping[bucket]
        label = mapping["label"]
        vad = {
            "valence": mapping["valence"],
            "arousal": mapping["arousal"],
            "dominance": mapping["dominance"],
        }
        medical_required = "13-16" == bucket

        # 2) 用 PSS-4 + 生理指标做修正 (PPG 低 + HR 高 → 升级到 自主神经活跃状态)
        if rmssd is not None and rmssd < 20 and (resting_hr is None or resting_hr > 85):
            if label != "自主神经活跃状态":
                label = "自主神经活跃状态"
                vad = {"valence": 2, "arousal": 8, "dominance": 2}
                medical_required = True

        # 3) 匹配模板
        ctx = {
            "pss4_score": pss4,
            "rmssd": rmssd if rmssd is not None else 0,
            "resting_hr": resting_hr if resting_hr is not None else 0,
            "stress_index": stress_idx if stress_idx is not None else 0,
        }

        matched_templates: list[str] = []
        for tpl in self.templates:
            cond = tpl.get("condition", "")
            try:
                if _eval_condition(cond, ctx):
                    matched_templates.append(tpl["templates"][0])
            except Exception:  # noqa: BLE001
                continue

        # 4) 置信度：基于指标数据完整度
        n_filled = sum(1 for v in (rmssd, resting_hr, stress_idx, pss4_score) if v is not None)
        confidence = 0.5 + 0.1 * n_filled
        if medical_required:
            confidence = max(0.7, confidence)

        return EmotionResult(
            label=label,
            vad=vad,
            templates=matched_templates,
            confidence=confidence,
            medical_required=medical_required,
        )
```

```python
"""医学告警引擎 - 消费 medical_alert.json 规则, 触发等级化告警。

设计依据: docs/planning/MVP1.0_技术方案_v1.1.md §17 (合规 / 告警体系)

告警等级:
    * green  正常         requires_action=False
    * yellow 关注         requires_action=False
    * orange 警示         requires_action=True
    * red    紧急         requires_action=True

每条规则有 confidence_required 阈值; 若评估 confidence 低于阈值,
规则不触发 (避免低质量数据引起误报)。
"""
from __future__ import annotations

import json
import logging
import re
from dataclasses import dataclass
from pathlib import Path
from typing import Any, Optional

from src.config import KNOWLEDGE_DIR
from src.utils.safe_condition_evaluator import SafeConditionEvaluator

logger = logging.getLogger(__name__)


@dataclass
class AlertResult:
    """单条医学告警结果。"""
    id: str
    level: str   # green / yellow / orange / red
    message: str
    medical_required: bool


class MedicalAlertEngine:
    """医学告警引擎。"""

    def __init__(self, config_path: Optional[Path] = None) -> None:
        self.config_path = config_path or (KNOWLEDGE_DIR / "medical_alert.json")
        if not self.config_path.exists():
            raise FileNotFoundError(f"medical_alert.json 不存在: {self.config_path}")
        with self.config_path.open(encoding="utf-8") as f:
            self.config = json.load(f)
        self.levels = self.config["alert_levels"]
        self.rules = self.config["alert_rules"]
        self._evaluator = SafeConditionEvaluator()

    def evaluate(
        self,
        indicators: dict[str, Any],
        user: dict[str, Any],
        confidence: float = 1.0,
    ) -> list[AlertResult]:
        """评估所有告警规则, 返回触发的告警列表 (按 level 倒序: red → orange → ...).

        Args:
            indicators: 标准指标字典
            user: 用户档案
            confidence: 整体测量置信度 (0-1), 低于规则阈值则规则不触发
        """
        ctx = self._build_context(indicators, user)
        triggered: list[AlertResult] = []

        for rule in self.rules:
            if confidence < rule.get("confidence_required", 0.0):
                continue

            # 把规则中的 'age_based_p10' / 'awgs_cutoff' 等抽象表达式替换为数值
            cond = self._substitute_context_refs(rule["condition"], ctx)

            try:
                if self._evaluator.evaluate(cond, ctx):
                    triggered.append(
                        AlertResult(
                            id=rule["id"],
                            level=rule["level"],
                            message=rule["message"],
                            medical_required=rule.get("medical_required", False),
                        )
                    )
            except Exception as e:  # noqa: BLE001
                logger.debug("规则 %s 评估失败: %s", rule["id"], e)

        # level 排序: red > orange > yellow > green
        severity_order = {"red": 0, "orange": 1, "yellow": 2, "green": 3}
        triggered.sort(key=lambda a: severity_order.get(a.level, 9))
        return triggered

    def _build_context(self, indicators: dict, user: dict) -> dict[str, Any]:
        """构造 condition 评估上下文 — 含 reference + age_based_p10 等。"""
        ctx: dict[str, Any] = dict(indicators)
        ctx.update(user)

        # 计算 age_based_p10 / p90 (HRV)
        age = user.get("age", 30)
        gender = user.get("gender", 1)
        rmssd = indicators.get("hrv_rmssd")

        # 简单查表: HRV P10 by age + gender
        p10_table = {
            (1, 25): 25.0, (1, 35): 22.0, (1, 55): 15.0,
            (0, 25): 22.0, (0, 35): 20.0, (0, 55): 14.0,
        }
        if rmssd is not None:
            # 近似查找最接近年龄段的 P10
            closest_age = min([25, 35, 55], key=lambda a: abs(a - age))
            ctx["age_based_p10"] = p10_table.get((gender, closest_age), 20.0)
            ctx["age_based_p90"] = 65.0  # 简化

        # body_fat_pct P90 (按年龄)
        body_fat = indicators.get("body_fat_pct")
        if body_fat is not None:
            if age <= 30:
                ctx["age_based_p90"] = 30.0
            elif age <= 50:
                ctx["age_based_p90"] = 32.0
            else:
                ctx["age_based_p90"] = 35.0

        # AWGS cutoff
        skm = indicators.get("skeletal_muscle_kg")
        smmi = indicators.get("smmi")
        if smmi is not None:
            ctx["awgs_cutoff"] = 7.0 if gender == 1 else 5.7
        elif skm is not None and indicators.get("height_cm"):
            try:
                from src.indicators.biometric_extractors import merge_indicators
                merged = merge_indicators({"skeletal_muscle_kg": skm}, {}, user)
                if merged.get("smmi") is not None:
                    ctx["awgs_cutoff"] = 7.0 if gender == 1 else 5.7
            except Exception:
                pass

        return ctx

    @staticmethod
    def _substitute_context_refs(condition: str, ctx: dict) -> str:
        """把规则 condition 中的上下文引用替换为字面量, 让 SafeConditionEvaluator 能求值。

        规则格式示例:
            "value < age_based_p10"
            "value > age_based_p90"
            "smmi < awgs_cutoff"
        把标识符替换为其值。
        """
        refs = {"age_based_p10", "age_based_p90", "awgs_cutoff"}
        for ref in refs:
            if ref in ctx and ref in condition:
                condition = condition.replace(ref, repr(ctx[ref]))
        return condition
```

```python
"""综合健康分计算引擎 - 消费 composite_health_scores.json。

设计依据: docs/planning/MVP1.0_技术方案_v1.1.md §15.3

5 个综合分:
    1. metabolic_composite     (代谢: BMI + 体脂率 + 内脏脂肪 + BMR)
    2. immune_composite        (免疫营养: 蛋白质率 + 水分率 + 体脂率)
    3. cardio_fitness_composite(心肺: HRV + HR + SpO2 + cardio_fitness)
    4. stress_composite        (压力: 压力指数 + HRV + PSS-4)
    5. bone_health_composite   (骨骼: 骨量)

计算策略:
    每项分 = sum( component_normalized_score * weight )
    component_normalized_score ∈ [0, 100]
    解释等级按 interpretation 字典匹配 (键为范围字符串)

声明: 全部为产品自定义计算, 非临床诊断。
"""
from __future__ import annotations

import json
import logging
from dataclasses import dataclass, field
from pathlib import Path
from typing import Any, Optional

from src.knowledge.medical_db import MedicalKnowledgeBase

logger = logging.getLogger(__name__)


@dataclass
class CompositeScore:
    """单条综合分计算结果。"""
    id: str
    name: str
    value: float
    interpretation: str
    components: dict[str, float] = field(default_factory=dict)
    note: str = ""


class CompositeScoreEngine:
    """综合分计算引擎。"""

    def __init__(self, config_path: Optional[Path] = None, knowledge_base: Optional[MedicalKnowledgeBase] = None) -> None:
        from src.config import KNOWLEDGE_DIR
        self.config_path = config_path or (KNOWLEDGE_DIR / "composite_health_scores.json")
        if not self.config_path.exists():
            raise FileNotFoundError(f"composite_health_scores.json 不存在: {self.config_path}")
        with self.config_path.open(encoding="utf-8") as f:
            self.config = json.load(f)
        self.score_definitions = self.config["scores"]
        self.kb = knowledge_base or MedicalKnowledgeBase()

    def calculate_all(
        self,
        indicators: dict[str, Any],
        user: dict[str, Any],
    ) -> list[CompositeScore]:
        """计算全部 5 个综合分, 返回 CompositeScore 列表。"""
        results: list[CompositeScore] = []
        for sd in self.score_definitions:
            try:
                results.append(self.calculate(sd["id"], indicators, user))
            except Exception as e:  # noqa: BLE001
                logger.warning("综合分 %s 计算失败: %s", sd["id"], e)
                results.append(
                    CompositeScore(
                        id=sd["id"], name=sd["name"], value=50.0,
                        interpretation="一般", note=f"计算失败: {e}",
                    )
                )
        return results

    def calculate(
        self,
        score_id: str,
        indicators: dict[str, Any],
        user: dict[str, Any],
    ) -> CompositeScore:
        """计算单条综合分。"""
        sd = next((s for s in self.score_definitions if s["id"] == score_id), None)
        if sd is None:
            raise ValueError(f"未定义综合分: {score_id}")

        weights = sd.get("weighting", {})
        components: dict[str, float] = {}
        weighted_sum = 0.0
        total_weight = 0.0

        for ind_id, weight in weights.items():
            value = indicators.get(ind_id)
            if value is None:
                continue
            normalized = self._normalize(ind_id, value, user, sd)
            components[ind_id] = round(normalized, 1)
            weighted_sum += normalized * weight
            total_weight += weight

        if total_weight > 0:
            value = round(weighted_sum / total_weight, 1)
        else:
            value = 50.0  # 兜底: 数据全缺时给中性值

        interpretation = self._interpret(value, sd.get("interpretation", {}))
        return CompositeScore(
            id=score_id,
            name=sd["name"],
            value=value,
            interpretation=interpretation,
            components=components,
            note=sd.get("note", "产品参考值, 非临床诊断"),
        )

    def _normalize(self, ind_id: str, value: Any, user: dict, sd: dict) -> float:
        """把单个指标归一化到 0-100 (score=100 = 最健康)."""
        # BMI 特殊处理 (目标区间 18.5-23.9 最健康)
        if ind_id == "bmi":
            bmi = float(value)
            if 18.5 <= bmi <= 23.9:
                return 100.0
            if 17 <= bmi < 18.5:
                return 80.0
            if 24 <= bmi <= 27.9:
                return 70.0
            if bmi >= 28 or bmi < 17:
                return 50.0
            return 60.0

        # 体脂率: 按参考范围判断 (male 标准 11-16.9%)
        if ind_id == "body_fat_pct":
            bf = float(value)
            age = user.get("age", 30)
            gender = user.get("gender", 1)
            ref = self.kb.get_reference("body_fat_pct", age=age, gender=gender)
            if ref and ref.get("low") is not None and ref.get("high") is not None:
                if ref["low"] <= bf <= ref["high"]:
                    return 90.0
                # 超出参考 越多 分越低
                deviation = abs(bf - (ref["low"] + ref["high"]) / 2)
                return max(50.0, 100.0 - deviation * 4)
            return 60.0

        # 内脏脂肪: <10 最佳
        if ind_id == "visceral_fat_level":
            v = int(value)
            if v < 10:
                return 95.0
            if v < 14:
                return 75.0
            if v < 18:
                return 50.0
            return 30.0

        # HRV RMSSD: 高 = 好 (粗略按线性)
        if ind_id == "hrv_rmssd":
            r = float(value)
            age = user.get("age", 30)
            gender = user.get("gender", 1)
            ref = self.kb.get_reference("hrv_rmssd", age=age, gender=gender)
            if ref and ref.get("median") is not None:
                median = ref["median"]
                return min(100.0, max(40.0, r / median * 100))
            return 60.0

        # HR: 60-100 最佳
        if ind_id == "heart_rate":
            hr = int(value)
            if 60 <= hr <= 80:
                return 95.0
            if 50 <= hr <= 100:
                return 75.0
            if 40 <= hr < 50 or 100 < hr <= 110:
                return 50.0
            return 30.0

        # SpO2: >=95 = 100, 92-94 = 中等, <92 = 差
        if ind_id == "spo2":
            s = int(value)
            if s >= 95:
                return 100.0
            if s >= 92:
                return 70.0
            return 30.0

        # 压力指数: 低 = 好
        if ind_id == "stress_index":
            s = float(value)
            if s < 30:
                return 95.0
            if s < 50:
                return 75.0
            if s < 70:
                return 55.0
            return 35.0

        # PSS-4
        if ind_id == "pss4_score":
            p = int(value)
            if p <= 4:
                return 95.0
            if p <= 8:
                return 75.0
            if p <= 12:
                return 50.0
            return 30.0

        # 蛋白质率: 14-18% 最佳
        if ind_id == "protein_pct":
            p = float(value)
            if 14 <= p <= 18:
                return 95.0
            return 70.0

        # 水分率: 50-65% 最佳
        if ind_id == "water_pct":
            w = float(value)
            if 50 <= w <= 65:
                return 95.0
            if 45 <= w <= 70:
                return 75.0
            return 55.0

        # BMR: 与标准值的偏差
        if ind_id == "bmr":
            return 70.0  # 中性值, 后续可优化

        # cardio_fitness: 已标准化 0-100
        if ind_id == "cardio_fitness":
            return min(100.0, max(0.0, float(value)))

        # bone_mass_kg: 与标准值相比
        if ind_id == "bone_mass_kg":
            b = float(value)
            if b >= 2.5:
                return 90.0
            if b >= 2.0:
                return 75.0
            return 55.0

        return 60.0  # 未知指标中性值

    @staticmethod
    def _interpret(value: float, ranges: dict[str, str]) -> str:
        """根据值落到 ranges 中匹配解释。"""
        # ranges 形如 {"0-30": "高压力", "30-50": "正常", ...}
        # 也可能带小数: "0-30.5"
        for range_key, label in ranges.items():
            try:
                if "-" in range_key:
                    lo, hi = range_key.split("-")
                    lo, hi = float(lo), float(hi)
                    if lo <= value <= hi:
                        return label
            except (ValueError, AttributeError):
                continue
        return "一般"
```

```python
"""报告渲染引擎 - 消费 report_templates/ JSON + indicators + advices + alerts + scores。

设计依据: docs/planning/MVP1.0_技术方案_v1.1.md §14 / §18 Step 6

12 条模板按 body_composition/pressure_assessment/cardio_fitness 等路由到对应内容生成器:
    * 每条模板定义 indicators + sections + disclaimer
    * 每条 section 有 type: 'indicator_table' | 'advice_list' | 'alert_list' | 'score_table' | 'text'
    * render_all() 返回 RenderResult 列表 (id / title / sections dict)

声明: 渲染结果由 ForbiddenWordsFilter 二次过滤, 确保不带禁用词。
"""
from __future__ import annotations

import json
import logging
from dataclasses import dataclass, field
from pathlib import Path
from typing import Any, Optional

from src.config import KNOWLEDGE_DIR
from src.knowledge.forbidden_words_filter import ForbiddenWordsFilter

logger = logging.getLogger(__name__)


@dataclass
class RenderResult:
    """单条模板渲染结果。"""
    id: str
    title: str
    sections: dict[str, str] = field(default_factory=dict)
    disclaimer: str = "本报告为消费级健康参考，非医疗器械。"


class ReportEngine:
    """报告模板渲染引擎。"""

    def __init__(self, templates_dir: Optional[Path] = None) -> None:
        self.templates_dir = templates_dir or (KNOWLEDGE_DIR / "report_templates")
        if not self.templates_dir.exists():
            raise FileNotFoundError(f"报告模板目录不存在: {self.templates_dir}")
        self._template_defs: dict[str, dict] = {}
        for tpl_path in sorted(self.templates_dir.glob("*.json")):
            with tpl_path.open(encoding="utf-8") as f:
                tpl = json.load(f)
            tpl["__path"] = str(tpl_path)
            self._template_defs[tpl_path.stem] = tpl
        self._filter = ForbiddenWordsFilter()
        logger.info("ReportEngine 加载 %d 条模板", len(self._template_defs))

    def render_all(
        self,
        indicators: dict[str, Any],
        user: dict[str, Any],
        advices: list,
        alerts: list,
        composite_scores: list,
    ) -> list[RenderResult]:
        """渲染全部模板, 返回 RenderResult 列表 (按模板 id 排序)."""
        results: list[RenderResult] = []
        for tid, tpl in sorted(self._template_defs.items()):
            try:
                results.append(self._render_one(tid, tpl, indicators, user, advices, alerts, composite_scores))
            except Exception as e:  # noqa: BLE001
                logger.warning("模板 %s 渲染失败: %s", tid, e)
                # 失败时仍返回空模板, 不阻断管线
                results.append(RenderResult(
                    id=tid,
                    title=tpl.get("title", tid),
                    sections={},
                    disclaimer="渲染失败, 数据可能不完整。",
                ))
        return results

    def _render_one(
        self, tid: str, tpl: dict, indicators: dict, user: dict,
        advices: list, alerts: list, scores: list,
    ) -> RenderResult:
        """渲染单条模板。"""
        sections_def = tpl.get("sections", [])
        rendered: dict[str, str] = {}

        # 1. 渲染每段
        for sec in sections_def:
            sec_type = sec.get("type", "text")
            sec_title = sec.get("title", "")
            content = ""
            if sec_type == "indicator_table":
                content = self._render_indicators(sec.get("indicators", []), indicators)
            elif sec_type == "advice_list":
                content = self._render_advices(advices, max_items=sec.get("max_items", 5))
            elif sec_type == "alert_list":
                content = self._render_alerts(alerts)
            elif sec_type == "score_table":
                content = self._render_scores(scores)
            elif sec_type == "text":
                content = sec.get("content", "")
            else:
                content = f"未知段类型: {sec_type}"

            if content:
                # 合规过滤
                content = self._filter.filter_text(content)
                rendered[sec_title or sec_type] = content

        # 2. 拼 disclaimer
        disclaimer_text = tpl.get("disclaimer") or (
            "本报告由消费级健康检测设备生成, 非医疗器械, "
            "不构成医疗诊断或改善建议。如有疑虑请咨询执业医师。"
        )
        disclaimer_text = self._filter.filter_text(disclaimer_text)

        return RenderResult(
            id=tid,
            title=tpl.get("title", tid),
            sections=rendered,
            disclaimer=disclaimer_text,
        )

    @staticmethod
    def _render_indicators(needed: list[str], indicators: dict) -> str:
        """渲染指标表格 — 显示 '指标名 = 数值 单位' 形式。"""
        rows = []
        for ind_id in needed:
            v = indicators.get(ind_id)
            if v is None:
                rows.append(f"  - {ind_id}: 数据缺失")
            else:
                rows.append(f"  - {ind_id}: {v}")
        return "\n".join(rows) if rows else "无相关数据"

    @staticmethod
    def _render_advices(advices: list, max_items: int = 5) -> str:
        """渲染建议列表 — 按 priority 排序。"""
        if not advices:
            return "暂无建议"
        priority_order = {"urgent": 0, "high": 1, "medium": 2, "low": 3}
        sorted_advices = sorted(advices, key=lambda a: priority_order.get(a.priority if hasattr(a, 'priority') else a.priority, 9))[:max_items]
        rows = []
        for a in sorted_advices:
            content = a.content if hasattr(a, 'content') else a.get('content', '')
            cat = a.category if hasattr(a, 'category') else a.get('category', '')
            rows.append(f"  [{a.priority}] {cat}: {content[:80]}")
        return "\n".join(rows)

    @staticmethod
    def _render_alerts(alerts: list) -> str:
        """渲染告警列表。"""
        if not alerts:
            return "未触发医学告警"
        rows = []
        for a in alerts:
            level = a.level if hasattr(a, 'level') else a.get('level', '?')
            msg = a.message if hasattr(a, 'message') else a.get('message', '')
            rows.append(f"  [{level}] {msg}")
        return "\n".join(rows)

    @staticmethod
    def _render_scores(scores: list) -> str:
        """渲染综合分表格。"""
        if not scores:
            return "综合分未计算"
        rows = []
        for s in scores:
            name = s.name if hasattr(s, 'name') else s.get('name', '')
            value = s.value if hasattr(s, 'value') else s.get('value', 0)
            interp = s.interpretation if hasattr(s, 'interpretation') else s.get('interpretation', '')
            rows.append(f"  - {name}: {value}/100 ({interp})")
        return "\n".join(rows)


__all__ = ["ReportEngine", "RenderResult"]

                    "heart_rate": 72, "spo2": 98, "stress_index": 45},
        user={"age": 30, "gender": 1},
        advices=[], alerts=[], composite_scores=[],
    )
    for r in results:
        print(f"\n=== {r.title} ===")
        for sec_title, content in r.sections.items():
            print(f"  [{sec_title}] {content[:80]}")
```

```python
"""大健康检测系统 — MVP1.0 主入口。

支持子命令:
    * scan-ports       — 扫描可用串口
    * cli-measure      — 命令行端到端跑通一次测量
    * gui-measure      — 启动 PyQt6 主窗口 + 测量向导
"""
from __future__ import annotations

import argparse
import sys
from pathlib import Path

PROJECT_ROOT = Path(__file__).resolve().parent.parent
if str(PROJECT_ROOT) not in sys.path:
    sys.path.insert(0, str(PROJECT_ROOT))

from src.utils.logger import get_logger, setup_logging
from src.config import load_config

logger = get_logger(__name__)


def cmd_scan_ports(_args):
    try:
        from serial.tools import list_ports
    except ImportError:
        logger.error("缺少 pyserial")
        return 2
    ports = list_ports.comports()
    if not ports:
        logger.info("未发现可用串口")
        return 0
    for p in ports:
        logger.info("  %s", p.device)
    return 0


def cmd_cli_measure(_args):
    from src.pipeline.health_assessment_pipeline import HealthAssessmentPipeline

    cfg = load_config()
    pipeline = HealthAssessmentPipeline.from_default()

    user = {
        "name": "CLI测试用户",
        "age": int(input("年龄: ")),
        "gender": int(input("性别 1男0女: ")),
        "height_cm": int(input("身高: ")),
        "weight_kg": float(input("体重: ") or 0),
    }
    report = pipeline.run(user, pss4_score=None)

    print("\n===== 测量报告 =====")
    print(f"情绪: {report.emotional_state} (conf={report.emotional_confidence:.2f})")
    print(f"风险等级: {report.risk_level}")
    print(f"建议 {len(report.advices)} 条:")
    for a in report.advices:
        print(f"  [{a.priority}] {a.rule_id}: {a.content[:60]}")
    print(report.disclaimer)
    return 0


def cmd_gui_measure(_args):
    """启动 PyQt6 主窗口 + 测量向导。"""
    try:
        from PyQt6.QtWidgets import QApplication
        from src.ui.main_window import MainWindow
    except ImportError as e:
        logger.error("缺少 PyQt6: %s", e)
        return 2

    cfg = load_config()
    app = QApplication(sys.argv)
    window = MainWindow(config=cfg)
    window.show()
    sys.exit(app.exec())


def build_parser():
    parser = argparse.ArgumentParser(prog="大健康检测系统 MVP1.0")
    sub = parser.add_subparsers(dest="command")

    sub.add_parser("scan-ports", help="扫描可用串口").set_defaults(func=cmd_scan_ports)
    sub.add_parser("cli-measure", help="命令行端到端测量").set_defaults(func=cmd_cli_measure)
    sub.add_parser("gui-measure", help="启动 GUI 主窗口").set_defaults(func=cmd_gui_measure)
    return parser


def main(argv=None):
    cfg = load_config()
    setup_logging(cfg.log_dir, level=cfg.log_level)
    parser = build_parser()
    args = parser.parse_args(argv)
    if not getattr(args, "func", None):
        parser.print_help()
        return 1
    return int(args.func(args) or 0)


if __name__ == "__main__":
    raise SystemExit(main())
```

```python
"""Page 4: 采集中 — 实时进度条 + 设备状态。"""
from __future__ import annotations

from PyQt6.QtCore import Qt, QTimer
from PyQt6.QtWidgets import (
    QLabel, QProgressBar, QPushButton, QTextEdit, QVBoxLayout, QWizardPage,
)


class CollectingPage(QWizardPage):
    """测量进度页 — 后台线程跑 Pipeline.run()。

    使用 ``QTimer.singleShot`` 异步触发, 完成后调用 ``wizard().next()``。
    """

    def __init__(self, parent=None) -> None:
        super().__init__(parent)
        self.setTitle("正在测量…")
        self.setSubTitle("数据采集中, 请保持姿势稳定")

        layout = QVBoxLayout()

        self.status_label = QLabel("准备中…")
        self.status_label.setAlignment(Qt.AlignmentFlag.AlignCenter)
        self.status_label.setStyleSheet("font-size: 14px; font-weight: bold;")
        layout.addWidget(self.status_label)

        self.progress = QProgressBar()
        self.progress.setRange(0, 100)
        self.progress.setValue(0)
        layout.addWidget(self.progress)

        self.log_text = QTextEdit()
        self.log_text.setReadOnly(True)
        self.log_text.setMaximumHeight(200)
        layout.addWidget(self.log_text)

        self.cancel_btn = QPushButton("取消")
        self.cancel_btn.clicked.connect(self._on_cancel)
        layout.addWidget(self.cancel_btn)

        self.setLayout(layout)

    def initializePage(self) -> None:
        """页面显示时启动测量。"""
        self.progress.setValue(10)
        self.status_label.setText("启动 BIA 测量…")
        self._append_log("BIA 模块初始化…")

        # 通过 wizard().property 取出用户档案
        wizard = self.wizard()
        user = {
            "name": wizard.field("name"),
            "age": int(wizard.field("age") or 30),
            "gender": int(wizard.field("gender") or 1),
            "height_cm": int(wizard.field("height_cm") or 170),
            "weight_kg": float(wizard.field("weight_kg") or 70),
        }
        wizard.setProperty("user", user)

        # 使用 250ms 模拟进度 (真实实现应通过 PipelineRunner 后台线程 signal)
        self._step = 10
        self._timer = QTimer(self)
        self._timer.timeout.connect(self._tick)
        self._timer.start(250)

    def _tick(self) -> None:
        self._step += 5
        self.progress.setValue(self._step)
        if self._step == 30:
            self.status_label.setText("BIA 测量完成, 启动 PPG…")
            self._append_log("BIA → 体脂率/肌肉量等指标获取")
        elif self._step == 60:
            self.status_label.setText("PPG 采集中…")
            self._append_log("PPG → 心率/SpO2/HRV 采集中")
        elif self._step == 90:
            self.status_label.setText("生成健康评估中…")
            self._append_log("评估引擎触发中…")
        elif self._step >= 100:
            self.progress.setValue(100)
            self.status_label.setText("✓ 测量完成")
            self._append_log("✓ 报告已生成")
            self._timer.stop()

            # 完成时: 把 report 注入到 wizard, 由 report_display_page 使用
            report = self._build_demo_report()
            self.wizard().setProperty("report", report)

            # 自动跳转到下一页
            QTimer.singleShot(800, lambda: self.wizard().next())

    def _build_demo_report(self) -> dict:
        """MVP1.0 demo: 构造一份假报告 (真实实现从 Pipeline 拿)."""
        return {
            "composite_score": 78,
            "emotion": "平静放松状态",
            "risk": "low",
            "advices": [
```

```python
from __future__ import annotations

from PyQt6.QtCore import Qt
from PyQt6.QtWidgets import QCheckBox, QLabel, QVBoxLayout, QWizardPage


class AlertConfirmPage(QWizardPage):
    """告警确认页。

    列出所有 urgent/orange/red 级别的告警, 用户必须逐条勾选确认已阅读。
    未确认时, "下一步" 按钮禁用。
    """

    def __init__(self, parent=None) -> None:
        super().__init__(parent)
        self.setTitle("医学告警确认")
        self.setSubTitle("请逐条阅读并确认以下提示")

        self._layout = QVBoxLayout()
        self.setLayout(self._layout)
        self.confirmations: list[QCheckBox] = []

    def initializePage(self) -> None:
        """每次显示时根据 report 动态构建 UI。"""
        # 清空旧控件
        while self._layout.count():
            item = self._layout.takeAt(0)
            if item.widget():
                item.widget().deleteLater()

        report = self.wizard().property("report") or {}
        alerts = report.get("alerts", [])

        if not alerts:
            label = QLabel("✓ 未触发医学告警")
            label.setStyleSheet("color: #4CAF50; font-size: 14px; font-weight: bold;")
            label.setAlignment(Qt.AlignmentFlag.AlignCenter)
            self._layout.addWidget(label)
            return

        for alert in alerts:
            text = f"[{alert.get('level', '?')}] {alert.get('message', '')}"
            cb = QCheckBox(text)
            cb.setStyleSheet("font-size: 12px; padding: 8px;")
            cb.stateChanged.connect(self._check_complete)
            self.confirmations.append(cb)
            self._layout.addWidget(cb)

        self._layout.addStretch()
        self._check_complete()

        tip = QLabel(
            "⚠️ 强烈建议勾选所有告警并咨询执业医师后再继续。"
            "本系统不构成医疗诊断, 仅供健康参考。"
        )
        tip.setStyleSheet("color: #F44336; font-size: 11px; padding: 8px;")
        tip.setWordWrap(True)
        self._layout.addWidget(tip)

    def _check_complete(self) -> None:
        """所有告警勾选后, '下一步' 按钮才可用。"""
        if not self.confirmations:
            self.completeChanged.emit()
            return
        all_checked = all(cb.isChecked() for cb in self.confirmations)
        self.setComplete(all_checked)
        self.completeChanged.emit()

    def isComplete(self) -> bool:
        if not self.confirmations:
            return True
        return all(cb.isChecked() for cb in self.confirmations)
```

```python
"""历史趋势图 — matplotlib 嵌入 QWidget。

支持:
    * 指标选择 (QComboBox)
    * 时间窗口 (近 7 / 30 / 90 天)
    * 折线图渲染 (FigureCanvasQTAgg)
    * 数值标签 + 参考范围带

from __future__ import annotations

import logging
from typing import Optional

from matplotlib.backends.backend_qtagg import FigureCanvasQTAgg
from matplotlib.figure import Figure
from PyQt6.QtCore import Qt
from PyQt6.QtWidgets import (
    QComboBox, QHBoxLayout, QLabel, QPushButton, QVBoxLayout, QWidget,
)

from src.database.db_manager import DBManager
from src.services.trend_query import TrendQuery

logger = logging.getLogger(__name__)


_INDICATOR_LABELS = {
    "body_fat_pct": "体脂率 (%)",
    "muscle_kg": "肌肉量 (kg)",
    "skeletal_muscle_kg": "骨骼肌 (kg)",
    "water_pct": "水分率 (%)",
    "bmi": "BMI",
    "heart_rate": "心率 (bpm)",
    "spo2": "血氧 (%)",
    "hrv_rmssd": "HRV (ms)",
    "visceral_fat_level": "内脏脂肪等级",
}


class TrendChartWidget(QWidget):
    """历史趋势图组件。"""

    def __init__(self, db_manager: Optional[DBManager] = None, parent=None) -> None:
        super().__init__(parent)
        self.db_manager = db_manager or DBManager()
        self.trend_query = TrendQuery(self.db_manager)

        self._build_ui()

    def _build_ui(self) -> None:
        layout = QVBoxLayout()

        # 控件行: 指标选择 + 时间窗口 + 刷新
        ctrl_row = QHBoxLayout()
        ctrl_row.addWidget(QLabel("指标:"))
        self.indicator_combo = QComboBox()
        self.indicator_combo.addItems(list(_INDICATOR_LABELS.values()))
        ctrl_row.addWidget(self.indicator_combo)

        ctrl_row.addWidget(QLabel("时间窗口:"))
        self.window_combo = QComboBox()
        self.window_combo.addItems(["近 7 天", "近 30 天", "近 90 天"])
        self.window_combo.setCurrentIndex(1)
        ctrl_row.addWidget(self.window_combo)

        refresh_btn = QPushButton("刷新")
        refresh_btn.clicked.connect(self.refresh)
        ctrl_row.addWidget(refresh_btn)
        ctrl_row.addStretch()

        layout.addLayout(ctrl_row)

        # matplotlib canvas
        self.figure = Figure(figsize=(8, 4), dpi=100)
        self.canvas = FigureCanvasQTAgg(self.figure)
        layout.addWidget(self.canvas)

        # 状态行
        self.status_label = QLabel("选择指标 + 点击刷新加载趋势")
        self.status_label.setStyleSheet("color: gray; font-size: 11px;")
        layout.addWidget(self.status_label)

        self.setLayout(layout)

    def refresh(self) -> None:
        """根据当前选择刷新图表。"""
        # 找到当前指标 id
        label_to_id = {v: k for k, v in _INDICATOR_LABELS.items()}
        current_label = self.indicator_combo.currentText()
        ind_id = label_to_id.get(current_label, "body_fat_pct")

        window_days = {0: 7, 1: 30, 2: 90}.get(self.window_combo.currentIndex(), 30)

        points = self.trend_query.get_trend(indicator_id=ind_id, window_days=window_days)

        self.figure.clear()
        ax = self.figure.add_subplot(111)
        if not points:
            ax.text(0.5, 0.5, "暂无历史数据", ha="center", va="center", transform=ax.transAxes, fontsize=14)
            self.status_label.setText(f"近 {window_days} 天无 {current_label} 测量记录")
        else:
            xs = [p["measured_at"] for p in points]
            ys = [p["value"] for p in points]
            ax.plot(xs, ys, marker="o", linewidth=2, color="#1976D2")
            ax.set_title(f"{current_label} — 近 {window_days} 天")
            ax.set_xlabel("测量时间")
            ax.set_ylabel(current_label)
            ax.grid(True, alpha=0.3)
            self.figure.autofmt_xdate()
            self.status_label.setText(f"已加载 {len(points)} 个数据点")

        self.canvas.draw()
```

```python
"""PipelineRunner — 在 QThread 中运行 HealthAssessmentPipeline。

通过 PyQt signal 与 UI 通信:
    * started
    * progress(str)         阶段性提示
    * finished(FinalReport) 测量完成, 携带完整报告
    * failed(str)           错误信息
"""
from __future__ import annotations

import logging
import traceback
from pathlib import Path
from typing import Optional

from PyQt6.QtCore import QThread, pyqtSignal
from PyQt6.QtWidgets import QApplication

from src.config import AppConfig
from src.pipeline.health_assessment_pipeline import HealthAssessmentPipeline
from src.pipeline.report_renderer import FinalReport

logger = logging.getLogger(__name__)


class PipelineRunner(QThread):
    """后台运行 HealthAssessmentPipeline。"""

    started = pyqtSignal()
    progress = pyqtSignal(str)           # 当前阶段描述
    finished = pyqtSignal(object)        # FinalReport
    failed = pyqtSignal(str)             # error message

    def __init__(
        self,
        user: dict,
        pss4_score: Optional[int],
        config: AppConfig,
        persist_to_db: bool = True,
        db_manager=None,
        bia_module=None,
        ppg_module=None,
    ) -> None:
        super().__init__()
        self.user = user
        self.pss4_score = pss4_score
        self.config = config
        self.persist_to_db = persist_to_db
        self.db_manager = db_manager
        self.bia_module = bia_module
        self.ppg_module = ppg_module
        self._cancelled = False

    def cancel(self) -> None:
        """请求取消 (graceful)."""
        self._cancelled = True

    def run(self) -> None:
        """QThread 入口 — 调用 pipeline.run() 并通过 signal 反馈。"""
        try:
            self.started.emit()
            self.progress.emit("初始化管线…")

            pipeline = HealthAssessmentPipeline(
                config=self.config,
                bia_module=self.bia_module,
                ppg_module=self.ppg_module,
                db_manager=self.db_manager,
            )

            # 注入进度回调 (当前实现: 后续可定制 stage hook)
            self.progress.emit("测量 BIA…")
            if self._cancelled:
                self.failed.emit("用户取消")
                return

            self.progress.emit("采集 PPG…")
            if self._cancelled:
                self.failed.emit("用户取消")
                return

            self.progress.emit("评估健康…")

            # Mock 化运行 (避免真实硬件依赖, 真实实现替换此处)
            report = pipeline.run(
                user=self.user,
                pss4_score=self.pss4_score,
                persist_to_db=self.persist_to_db,
                user_id=self._resolve_user_id() if self.persist_to_db else None,
            )

            self.progress.emit("生成报告…")
            self.finished.emit(report)

        except Exception as e:  # noqa: BLE001
            logger.exception("PipelineRunner 异常")
            self.failed.emit(f"{e}\n{traceback.format_exc()}")

    def _resolve_user_id(self) -> Optional[int]:
        """根据 user dict 查找或创建用户记录。"""
        if self.db_manager is None:
            return None
        from src.database.models import User

        u = User(
            name=self.user.get("name", "测试用户"),
            age=int(self.user.get("age", 30)),
            gender=int(self.user.get("gender", 1)),
            height_cm=int(self.user.get("height_cm", 170)),
        )
        try:
            return self.db_manager.get_or_create_user(u)
        except Exception as e:  # noqa: BLE001
            logger.warning("用户解析失败: %s", e)
            return None


__all__ = ["PipelineRunner"]
```

```python
"""数据导出服务 — JSON / CSV。

MVP1.0 支持:
    * export_latest_measurement(format)        — 导出最近一次测量
    * export_user_history(user_id, format)    — 导出指定用户全部测量
"""
from __future__ import annotations

import csv
import json
import logging
from datetime import datetime
from pathlib import Path
from typing import Any, Optional

from src.config import EXPORT_DIR
from src.database.db_manager import DBManager

logger = logging.getLogger(__name__)


class ExportService:
    """数据导出服务。"""

    def __init__(self, db_manager: DBManager, export_dir: Optional[Path] = None) -> None:
        self.db = db_manager
        self.export_dir = export_dir or EXPORT_DIR
        self.export_dir.mkdir(parents=True, exist_ok=True)

    def export_latest_measurement(self, format: str = "json") -> Path:
        """导出最近一次测量报告, 返回文件路径。"""
        latest = self.db.list_all_measurements(limit=1)
        if not latest:
            raise ValueError("无历史测量, 无法导出")
        m = latest[0]
        user = self.db.get_user(m.user_id)
        advice_list = self.db.list_advice(m.id)
        return self._save(m, user, advice_list, format=format, suffix="latest")

    def export_user_history(self, user_id: int, format: str = "json") -> Path:
        """导出指定用户全部历史测量。"""
        measurements = self.db.list_measurements(user_id=user_id, limit=1000)
        if not measurements:
            raise ValueError(f"用户 #{user_id} 无历史测量")
        user = self.db.get_user(user_id)
        out: list[dict[str, Any]] = []
        for m in measurements:
            advice_list = self.db.list_advice(m.id)
            out.append(self._build_payload(m, user, advice_list))
        return self._save_list(out, format=format, suffix=f"user_{user_id}_history")

    # ------------------------------------------------------------------ #
    # 内部工具
    # ------------------------------------------------------------------ #

    def _save(self, m, user, advice_list, format: str, suffix: str) -> Path:
        payload = self._build_payload(m, user, advice_list)
        return self._write(payload, format=format, suffix=suffix)

    def _save_list(self, items: list[dict], format: str, suffix: str) -> Path:
        if format == "json":
            return self._write(items, format="json", suffix=suffix)
        # CSV: flatten to single table
        if not items:
            raise ValueError("空数据, 无法导出 CSV")
        flat_rows = []
        for entry in items:
            row = {
                "user_name": entry.get("user", {}).get("name"),
                **{k: v for k, v in entry.get("measurement", {}).items()},
                "advices": "; ".join(a.get("content", "") for a in entry.get("advices", [])),
            }
            flat_rows.append(row)
        return self._write({"measurements": flat_rows}, format="csv", suffix=suffix)

    @staticmethod
    def _build_payload(m, user, advice_list) -> dict[str, Any]:
        m_dict = {k: (str(v) if hasattr(v, "isoformat") else v) for k, v in m.__dict__.items()}
        u_dict = {k: (str(v) if hasattr(v, "isoformat") else v) for k, v in user.__dict__.items()} if user else {}
        return {
            "exported_at": datetime.now().isoformat(timespec="seconds"),
            "user": u_dict,
            "measurement": m_dict,
            "advices": [
                {"id": a.id, "category": a.category, "priority": a.priority, "content": a.content}
                for a in advice_list
            ],
        }

    def _write(self, data, format: str, suffix: str) -> Path:
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        if format == "json":
            path = self.export_dir / f"{suffix}_{timestamp}.json"
            path.write_text(json.dumps(data, ensure_ascii=False, indent=2, default=str), encoding="utf-8")
        elif format == "csv":
            path = self.export_dir / f"{suffix}_{timestamp}.csv"
            if isinstance(data, dict) and "measurements" in data:
                rows = data["measurements"]
            elif isinstance(data, list):
                rows = data
            else:
                # single export: flat one row
                row = {
                    "user_name": data.get("user", {}).get("name"),
                    **{k: v for k, v in data.get("measurement", {}).items()},
                    "advices": "; ".join(a.get("content", "") for a in data.get("advices", [])),
                }
                rows = [row]
            if rows:
                with path.open("w", newline="", encoding="utf-8-sig") as f:
                    writer = csv.DictWriter(f, fieldnames=list(rows[0].keys()))
                    writer.writeheader()
                    writer.writerows(rows)
        else:
            raise ValueError(f"不支持的格式: {format}")
        logger.info("已导出: %s", path)
        return path


__all__ = ["ExportService"]
```

```python
from typing import Any, Optional

from src.database.db_manager import DBManager

logger = logging.getLogger(__name__)


class TrendQuery:
    """趋势查询服务。"""

    def __init__(self, db_manager: DBManager) -> None:
        self.db = db_manager

    def get_trend(
        self,
        indicator_id: str,
        window_days: int = 30,
        user_id: Optional[int] = None,
    ) -> list[dict[str, Any]]:
        """返回 (measured_at, value) 点列表, 按时间正序。

        Args:
            indicator_id: 医学指标 id (如 'body_fat_pct')
            window_days: 时间窗口 (天)
            user_id: 用户 id (None = 所有用户的最近测量)
        """
        cutoff = (datetime.now() - timedelta(days=window_days)).isoformat(timespec="seconds")
        with self.db.connection() as conn:
            if user_id is not None:
                rows = conn.execute(
                    """SELECT measured_at, body_fat_pct, water_pct, muscle_kg,
                              skeletal_muscle_kg, bmr_kcal, bone_kg,
                              visceral_fat_level, body_age, weight_kg,
                              protein_pct, bmi,
                              heart_rate, spo2, perfusion_index, hrv_rmssd
                       FROM measurements
                       WHERE user_id = ? AND measured_at >= ?
                       ORDER BY measured_at ASC""",
                    (user_id, cutoff),
                ).fetchall()
            else:
                rows = conn.execute(
                    """SELECT measured_at, body_fat_pct, water_pct, muscle_kg,
                              skeletal_muscle_kg, bmr_kcal, bone_kg,
                              visceral_fat_level, body_age, weight_kg,
                              protein_pct, bmi,
                              heart_rate, spo2, perfusion_index, hrv_rmssd
                       FROM measurements
                       WHERE measured_at >= ?
                       ORDER BY measured_at ASC""",
                    (cutoff,),
                ).fetchall()

        points: list[dict[str, Any]] = []
        for row in rows:
            d = dict(row)
            value = d.get(indicator_id)
            if value is None:
                continue
            points.append({
                "measured_at": d["measured_at"],
                "value": float(value),
            })
        return points

    def get_latest_measurement_with_advice(self) -> Optional[dict[str, Any]]:
        """返回最近一次测量 + 建议的完整 dict。"""
        latest = self.db.list_all_measurements(limit=1)
        if not latest:
            return None
        m = latest[0]
        user = self.db.get_user(m.user_id)
        advice_list = self.db.list_advice(m.id)
        return {
            "user_id": m.user_id,
            "measured_at": m.measured_at,
            "measurement": {
                k: (str(v) if hasattr(v, "isoformat") else v)
                for k, v in m.__dict__.items() if k != "id"
            },
            "user": {k: v for k, v in user.__dict__.items()} if user else {},
            "advice": [
                {"id": a.id, "category": a.category, "priority": a.priority, "content": a.content}
                for a in advice_list
            ],
            "indicators": {
                k: getattr(m, k) for k in (
                    "body_fat_pct", "muscle_kg", "water_pct", "bmi",
                    "heart_rate", "spo2", "hrv_rmssd",
                ) if getattr(m, k, None) is not None
            },
            "composite_score": getattr(m, "health_score", None),
            "risk_level": getattr(m, "risk_level", "low"),
        }


__all__ = ["TrendQuery"]
```

```python
"""PyInstaller 打包脚本 — 产出 dist/HealthMvp1.0.exe。

使用:
    python scripts/build_pyinstaller.py

依赖:
    pyinstaller>=6.0  (requirements.txt 已声明)
"""
from __future__ import annotations

import argparse
import shutil
import sys
from pathlib import Path

PROJECT_ROOT = Path(__file__).resolve().parent.parent


def build(onefile: bool = True, console: bool = False) -> int:
    """调用 PyInstaller.main() 构造 exe。

    Args:
        onefile: 单文件 vs 单目录
        console: 是否显示控制台 (GUI 程序设为 False)
    """
    try:
        import PyInstaller.__main__ as pyi_main
    except ImportError:
        print("ERROR: 缺少 pyinstaller. 运行: pip install pyinstaller>=6.0")
        return 1

    args = [
        str(PROJECT_ROOT / "src" / "main.py"),
        "--name=HealthMvp1.0",
        "--paths=" + str(PROJECT_ROOT),
        "--add-data=" + str(PROJECT_ROOT / "data") + os_sep() + "data",
        "--add-data=" + str(PROJECT_ROOT / "config.yaml") + os_sep() + ".",
        "--hidden-import=PyQt6.QtCore",
        "--hidden-import=PyQt6.QtGui",
        "--hidden-import=PyQt6.QtWidgets",
        "--hidden-import=matplotlib.backends.backend_qtagg",
        "--collect-submodules=src",
        "--clean",
        "--noconfirm",
    ]

    if onefile:
        args.append("--onefile")
    else:
        args.append("--onedir")

    if console:
        args.append("--console")
    else:
        args.append("--noconsole")  # GUI 模式隐藏控制台

    print("=== PyInstaller 参数 ===")
    for a in args:
        print(f"  {a}")
    print(f"开始打包 (onefile={onefile}, console={console})…")

    return pyi_main.run(args)


def os_sep() -> str:
    """PyInstaller --add-data 路径分隔符 (Windows 是 `;` / Unix 是 `:`)."""
    import os
    return ";" if os.name == "nt" else ":"


def main() -> int:
    parser = argparse.ArgumentParser()
    parser.add_argument("--onefile", action="store_true", default=True, help="单文件模式 (默认)")
    parser.add_argument("--onedir", action="store_true", help="单目录模式")
    parser.add_argument("--console", action="store_true", help="显示控制台 (debug 用)")
    args = parser.parse_args()

    onefile = not args.onedir
    return build(onefile=onefile, console=args.console)


if __name__ == "__main__":
    raise SystemExit(main())
```
